\syma{AMBA}{\textit{Advanced Microcontroller Bus Architecture} -- Arquitectura de bus de microcontrolador avanzado}

\syma{ASIC}{\textit{Application-Specific Integrated Circuit} -- Circuito integrado de aplicación específica}

\syma{AXI}{\textit{Advanced eXtensible Interface} -- Interfaz extensible avanzada}

\syma{BER}{\textit{Bit Error Rate} -- Tasa de error de bit}

\syma{BRAM}{\textit{Block RAM} -- Bloque de memoria RAM dentro de la FPGA}

\syma{BSP}{\textit{Board Support Package} -- Paquete de soporte de tarjeta}

\syma{CRP}{\textit{Challenge-Response Pair} -- Par desafío-respuesta}

\syma{DRC}{\textit{Design Rule Check} -- Verificación de reglas de diseño}

\syma{DSP}{\textit{Digital Signal Processor} -- Procesador digital de señales}

\syma{FAR}{\textit{False Acceptance Rate} -- Tasa de falsa aceptación}

\syma{FF}{\textit{Flip-Flop} -- Biestable}

\syma{FPGA}{\textit{Field Programmable Gate Array} -- Arreglo de compuertas programable en campo}

\syma{FRR}{\textit{False Rejection Rate} -- Tasa de falso rechazo}

\syma{FSM}{\textit{Finite State Machine} -- Máquina de estados finitos}

\syma{HD}{\textit{Hamming Distance} -- Distancia de Hamming}

\syma{HDA}{\textit{Helper Data Algorithm} -- Algoritmo de datos auxiliares}

\syma{HDL}{\textit{Hardware Description Language} -- Lenguaje de descripción de hardware}

\syma{IoT}{\textit{Internet of Things} -- Internet de las cosas}

\syma{IP}{\textit{Intellectual Property} -- Propiedad intelectual (núcleo IP)}

\syma{ISA}{\textit{Instruction Set Architecture} -- Arquitectura del conjunto de instrucciones}

\syma{JTAG}{\textit{Joint Test Action Group} -- Estándar IEEE 1149.1 de depuración}

\syma{LFSR}{\textit{Linear Feedback Shift Register} -- Registro de desplazamiento con retroalimentación lineal}

\syma{LMB}{\textit{Local Memory Bus} -- Bus local de memoria}

\syma{LUT}{\textit{Look-Up Table} -- Tabla de búsqueda}

\syma{MDM}{\textit{MicroBlaze Debug Module} -- Módulo de depuración de MicroBlaze}

\syma{MMIO}{\textit{Memory-Mapped I/O} -- Entrada/salida mapeada en memoria}

\syma{NCR}{\textit{Number of CRPs} -- Número de pares desafío-respuesta requeridos para modelar una PUF}

\syma{NLFSR}{\textit{Non-Linear Feedback Shift Register} -- Registro de desplazamiento con retroalimentación no lineal}

\syma{NLPUF}{\textit{Non-Linear PUF} -- PUF basada en lógica no lineal}

\syma{PLL}{\textit{Phase-Locked Loop} -- Lazo de fase enganchada}

\syma{PLPUF}{\textit{Pseudo-LFSR PUF} -- PUF basada en pseudo-LFSR}

\syma{PRNG}{\textit{Pseudo-Random Number Generator} -- Generador de números pseudoaleatorios}

\syma{PUF}{\textit{Physical Unclonable Function} -- Función físicamente no clonable}

\syma{RISC-V}{\textit{Reduced Instruction Set Computer~V} -- Arquitectura abierta de conjunto de instrucciones reducido}

\syma{RO-PUF}{\textit{Ring Oscillator PUF} -- PUF basada en oscilador de anillo}

\syma{SoC}{\textit{System-on-Chip} -- Sistema en chip}

\syma{SRAM}{\textit{Static Random Access Memory} -- Memoria estática de acceso aleatorio}

\syma{TMR}{\textit{Triple Modular Redundancy} -- Redundancia modular triple}

\syma{TTM}{\textit{Time to Model} -- Tiempo computacional para modelar una PUF}

\syma{UART}{\textit{Universal Asynchronous Receiver-Transmitter} -- Receptor y transmisor asíncrono universal}

\syma{XDC}{\textit{Xilinx Design Constraints} -- Archivo de restricciones de diseño}

\syma{XOR}{\textit{Exclusive OR} -- Compuerta lógica O-exclusiva}

\syma{XSA}{\textit{Xilinx Support Archive} -- Archivo de soporte de Xilinx}
